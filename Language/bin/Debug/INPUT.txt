func Sum(a, b){
	return (a+b);
}
func Multipy(a, b){
	return (a*b);
}
func Pow(a){
	return (a*a);
}

func Main()
{
	A = 2;
	Display(A);
	B = 3;
	Display(B);
	summa = Sum(A, B);
	Display(summa);
	multiplication = Multipy(A, B);
	Display(multiplication);
	powA = Pow(A);
	Display(powA);
	powB = Pow(B);
	Display(powB);
	twoAB = 2*multiplication;
	Display(twoAB);
	tempSum = Sum(powA, twoAB);
	Display(tempSum);
	sqSum = Sum(tempSum, powB);
	Display(sqSum);

	sqSumShort = Pow(summa);
	Display(sqSumShort);
	

	return sqSumShort;	
}
